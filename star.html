1. 별찍기의 기본원리

별찍기는 컴퓨터 모니터의 좌표와 좌표계에서 함수의 부등식 영역을 이해하면 쉽게 그릴 수 있다. (이는 모든 프로그램 언어에 공히 적용된다.) 컴퓨터 모니터의 좌표는 행렬의 인덱스와 같다.


< 모니터의 좌표계>
0, 0 0, 1 0, 2 0, 3 0, 4
1, 0 1, 1 1, 2 1, 3 1, 4
2, 0 2, 1 2, 2 2, 3 2, 4
3, 0 3, 1 3, 2 3, 3 3, 4
4, 0 4, 1 4, 2 4, 3 4, 4

이는 우리가 수학시간에 배우는 데카르트 좌표계와 x축 방향은 같고 y축 방향은 반대임을 알 수 있다. 따라서 우리가 찍으려는 별모양의 x축 대칭인 함수형태를 구상하면 된다.

   

그림에서 4/4 분면이 모니터의 좌표계이다. 그래서 <그림 1>에서 회색영역인 왼쪽붙은 직각삼각형을 출력하려면 x축에 대칭인 1/4 분면의 함수부등식 y > x ( <그림 1>에서 분홍색 영역)를 고려하면 된다. 코드에서 행을 y, 열을 x로 표시한다. 일단 모든 y, x에 대해 0부터 n 까지 루프를 돌린다. 그리고 y > x 인 경우 별을 찍으면 된다. 프로그램 언어마다 조금씩 차이는 있지만 자바스크립트의 경우는 다음과 같다.

<그림 2>에서 회색영역인 왼쪽붙은 역직각삼각형을 출력하려면 x축에 대칭인 1/4 분면의 함수부등식 y < n - x 을 써주면 된다.

이 논리를 적용하면 피라미드형, 역피라미드형, 다이아몬드형, 모래시계형 모양을 찍을 수 있다. 심지어 피라미드 2개를 나라히 찍을 수도 있다. x, y를 구간으로 나눠 구간별로 부등식을 적용하면 된다. 자세한 내용은 브라우저에서 웹페이지 소스 보기를 참조하기 바란다.


    // 피라미드 2개 나란히 찍기
       function starPrint8() {
        var n;
        n = parseInt(document.getElementById("input").value);
        var table = "<table id = table2>";
        for (var y = n; y > 0; y--) {
            table += "<tr>";
            for (var x = -2 * n; x < 2 * n; x++) {
                if (is2Pyramid(n, x, y))
                    table += "<td id = td2>*</td>";
                else
                    table += "<td id = td2></td>";
            }
            table += "</tr>";
        }
        table += "</table>";
        document.getElementById("output").innerHTML = "<피라미드 2개 나란히 찍기><br><br>" + table;

        // 피라미드 2개의 영역을 판정하는 함수
        function is2Pyramid(n, x, y) {
            if (x <= -n) {
                if (y <= x + 2 * n)
                    return true;
                else
                    return false;
            }
            else if (x >= -n && x <= 0) {
                if (y <= -x)
                    return true;
                else
                    return false;
            }
            else if (x >= 0 && x <= n) {
                if (y <= x)
                    return true;
                else
                    return false;
            }
            else {
                if (y <= 2 * n - x)
                    return true;
                else
                    return false;
            }
        }
    }

피라미드를 아래 위로 여러 개를 연달아 찍으려면 어떻게 해야 할까? 피라미드의 갯수가 늘어날수록 위와 같이 일일이 구간을 정하여 고려하기가 매우 복잡해진다. 이 경우는 수학시간에 배운 평행이동의 개념을 적용할 수 있다. y축으로 a만큼 평행이동하려면 y에 y-a를 대입하고 x축으로 b만큼 평행이동하려면 x에 x-b를 대입하면 된다. 

    
        function starPrint10() {
            var n;
            var arr = [[],[]];
            n = parseInt(document.getElementById("input").value);
            
            // 2n 개 행을 갖는 배열 생성
            for(var y = 4*n-1; y >= 0; y-- ) {
                arr.push([]);
            }
            
            // 4n 개 열을 갖는 배열 생성
            for (var y = 4*n-1; y >= 0; y--){
                for (var x =  0; x < 8*n; x++){
                    arr[y].push(" ");
                }
            }
            
            // 파리미드형 배열을 생성하는 함수
            function pyramid(n, a, b){ // y는 a 만큼 이동, x는 b 만큼 이동
                for(var y = n + a - 1; y >= 0 + a; y--){
                    for(var x = b; x < 2*n + b; x++){
                        if (x <= n + b && y < x - b + a)
                            arr[y][x] = '*';
                        else if (x > n + b && y < -(x - b) + 2*n + a)
                            arr[y][x] = '*';
                        else
                            arr[y][x] = ' ';
                    }
                }
            }
            
            pyramid(n, 0, 0);
            pyramid(n, 0, 2*n);
            pyramid(n, n, 0);
            pyramid(n, n, 2*n);
            pyramid(n, n, 4*n);

            // 배열을 스트링으로 만들어서 찍기           
            var table = "<table id = table2>";
            
            for (var y = 2*n-1; y >= 0; y--) {
                table += "<tr>";
                for (var x = 0; x < 8*n; x++) {
                    table += "<td id = td2>" + arr[y][x] + "</td>";
                }
                table += "</tr>";
            }
            table += "</table>";
            
            document.getElementById("output").innerHTML = "<n개 피라미드형 별찍기><br><br>" + table;
            }
