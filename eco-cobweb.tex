\tikzmath
{
function cobweb(\a,\b,\c,\d,\x0,\n) % \x0 는 초기값, \n은 반복횟수
  {
   function de(\x) % 수요함수
   {
   return \a+\b*\x;
   };
    function su(\x) % 공급함수
   {
   return \c+\d*\x;
   };
   function si(\x) % 공급함수 역함수
   {
   return (\x-\c)/\d;
   };
   { 
\draw[->](0,0)--(0,8) node[above]{P(가격)};
 \draw[->](0,0)--(10,0) node[right]{Q(수량)};
    \clip (-1,-1) rectangle (11,8);
   \draw[color=blue,domain=0:8] plot (\x,{de(\x)});    
    \draw[color=blue,domain=0:8] plot (\x,{su(\x)}); 
   \node at (7,6) {공급곡선};
  \node at (7,1) {수요곡선};
  \node at (-0.2,-0.2) {0};
  \newcommand{\start}{\x0} %초기값 설정
\pgfmathsetmacro{\xa}{\start} 
\pgfmathsetmacro{\ya}{de(\xa)}
\pgfmathsetmacro{\xb}{si(\ya)}
 \pgfmathsetmacro{\yb}{de(\xb)}
  \node at (\xa,-0.2) {$\start$};
\draw[->,color=magenta](\xa,0)--(\xa,\ya);
\draw[->,color=magenta](\xa,\ya)  edge (\xb,\ya) (\xb,\ya) edge (\xb,\yb);
%\dd \dd  \dd  \dd  \dd  \dd  \dd  \dd
\Repeat{\n}{\dd}
};
};
function fluc(\a,\b,\c,\d,\x0,\n) % x0 는 초기값, \n은 반복횟수
  {
   function de(\x) % 수요함수
   {
   return \a+\b*\x;
   };
    function su(\x) % 공급함수
   {
   return \c+\d*\x;
   };
   function si(\x) % 공급함수 역함수
   {
   return (\x-\c)/\d;
   };
   {
    \draw[->](0,0)--(0,6) node[above]{P(가격)};
    \draw[->](0,0)--(10,0) node[right]{t(시간)};
    \clip (-1,-1) rectangle (11,7);
\node at (-0.2,-0.2) {0};
\newcommand{\start}{\x0} %초기값 설정
\pgfmathsetmacro{\xa}{\start} 
\pgfmathsetmacro{\ya}{de(\xa)}
\pgfmathsetmacro{\xb}{si(\ya)}
 \pgfmathsetmacro{\yb}{de(\xb)}
 \draw [color=magenta] (1, \ya)--(2, \yb);
 \foreach \i in {2,...,\n}
{
\pgfmathparse{\xb}
 \xdef\xa{\pgfmathresult}
 \pgfmathparse{\yb}
 \xdef\ya{\pgfmathresult}
\pgfmathparse{si(\ya)}
 \xdef\xb{\pgfmathresult}
\pgfmathparse{de(\xb)}
 \xdef\yb{\pgfmathresult}
 \draw [color=magenta] (\i, \ya)--(\i+1, \yb);
}
 };
 };
}
